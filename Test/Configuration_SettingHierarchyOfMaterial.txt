    \    Scroll Web To Element    xpath=//*[@id="gridMaterialTier"]/table/tbody/tr[${index}]/td[1]
    \    ${name}    Get Web Text    xpath=//*[@id="gridMaterialTier"]/table/tbody/tr[${index}]/td[1]
    \    Run Keyword If    "${name}" == "${tiername}"    Exit For Loop
    Capture ScreenShot
